% Slide 2: Architecture Overview
\begin{frame}
  \frametitle{Clean Layered Architecture Enforces Separation}
  \begin{itemize}
    \item API layer handles HTTP requests and validation
    \item Services encapsulate business rules and orchestration
    \item Repositories abstract data access and SQL queries
    \item Unidirectional flow prevents coupling and side effects
  \end{itemize}

% Slide 3: Technology Stack
\begin{frame}
  \frametitle{Modern Stack Enables Performance and Safety}
  \begin{itemize}
    \item Frontend: React, TypeScript, Vite toolchain
    \item Backend: Rust, Actix-web, SQLx ORM
    \item Database: PostgreSQL with partitioning strategy
    \item Caching: Redis for real-time state and speed
  \end{itemize}
\end{frame}

% Slide 3: Core Services

    \item \textbf{Assignment Service:} Prevent scheduling conflicts, track history
    \item \textbf{Maintenance Service:} Schedule by mileage or time, auto-alerts

% Slide 6: Type Safety and Validation
\begin{frame}
  \frametitle{Type Safety and Compile-Time Validation Prevent Runtime Errors}
  
  \begin{block}{Safety Guarantees}
    \begin{itemize}
      \item Rust guarantees memory safety without garbage collection
      \item SQLx validates SQL queries at compile time, not runtime
      \item TypeScript frontend eliminates entire class of JavaScript errors
      \item Generated API client ensures frontend-backend contract
    \end{itemize}
  \end{block}
  
  \vfill
  \textbf{Benefits:}
  \begin{itemize}
    \item 70\% fewer production runtime errors from type mismatches
    \item Confident refactoring with compiler-enforced changes
    \item API breaking changes caught during build, not in production
  \end{itemize}
\end{frame}
